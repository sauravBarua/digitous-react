import React from 'react';
import 'bootstrap/dist/css/bootstrap.min.css';


class App extends React.Component {
  constructor(props){
    super(props)
      this.State = {
        value:'',
        email:'',
        password:''
    }
  }

  onChangeUserEmail(event){
    this.setState({email: event.target.value})
  }

  onChangeUserPassword(event){
    this.setState({password: event.target.value})
  }
 
    render() {
      return (
        <div>
          <div className="container">
          <form>
            <div className="form-group">
              <label> Email address:</label>
              <input type="email" className="form-control" value= {this.state.email} onChange = {this.onChangeUserEmail.bind(this)} />
            </div>
            <div className="form-group">
  
              <label>Password:</label>
              <input type="password" className="form-control" value= {this.state.password} onChange = {this.onChangeUserPassword.bind(this)}  />
            </div>
            <div>
              <label>Remember me:</label>
              <input type="checkbox" />
  
            </div>
            <div className="form-group">
              <button type="submit" className="btn btn-primary">Submit</button>
            </div>
  
          </form>
          </div>
        </div>
      )
    }
  }
  export default App;
